7.1: Playwright Basics

Playwright is a modern E2E testing framework by Microsoft. It supports Chromium, Firefox, and WebKit browsers.

Installation

npm install --save-dev @playwright/test npx playwright install # Create tests folder and first test file npx playwright codegen https://example.com # Record tests interactively

7.2: First E2E Test

// login.spec.js const { test, expect } = require('@playwright/test'); test('should login successfully', async ({ page }) => { // Navigate to website await page.goto('https://example.com/login'); // Fill form await page.fill('input[type="email"]', 'user@test.com'); await page.fill('input[type="password"]', 'password123'); // Click button await page.click('button:has-text("Login")'); // Wait for navigation await page.waitForURL('/dashboard'); // Assert await expect(page).toHaveTitle(/Dashboard/); }); // Run: npx playwright test // Run in headed mode: npx playwright test --headed // Debug: npx playwright test --debug

7.3: Selectors & Actions

Locators

// Selectors page.locator('#id') // ID page.locator('.class') // Class page.locator('button') // Tag page.locator('text=Click me') // Text content page.locator('button:has-text("Login")') // Text matcher page.locator('//button[@id="submit"]') // XPath // Actions await page.goto(url) // Navigate await page.click('#button') // Click await page.fill('#input', 'text') // Type await page.selectOption('#select', 'value') // Select dropdown await page.hover('#element') // Hover await page.press('#input', 'Enter') // Press key await page.screenshot({path: 'screenshot.png'}) // Screenshot

7.4: Waiting Strategies

// Auto-waiting (Playwright waits automatically) await page.click('#button'); // Waits for button to appear and be ready // Explicit waits await page.waitForURL('/dashboard'); // Wait for navigation await page.waitForSelector('#loader', { state: 'hidden' }); // Wait for element await page.waitForTimeout(2000); // Wait 2 seconds // Assertions (auto-waiting) await expect(page.locator('#message')).toBeVisible(); await expect(page.locator('#count')).toContainText('5');

7.5: Real Project - Shopping Flow

// shopping.spec.js const { test, expect } = require('@playwright/test'); test('complete shopping flow', async ({ page }) => { // Go to site await page.goto('https://shop.example.com'); // Search product await page.fill('input[placeholder="Search"]', 'laptop'); await page.press('input[placeholder="Search"]', 'Enter'); // Add to cart await page.click('text=Laptop Pro >> button:has-text("Add")'); // Go to checkout await page.click('button:has-text("Checkout")'); // Fill shipping await page.fill('#address', '123 Main St'); await page.fill('#city', 'New York'); // Select payment await page.selectOption('#payment', 'credit-card'); // Place order await page.click('button:has-text("Place Order")'); // Verify success await expect(page).toHaveURL(/order-confirmation/); await expect(page.locator('.success-message')).toBeVisible(); });
