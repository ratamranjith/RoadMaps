4.1: Jest Fundamentals

Jest is a JavaScript testing framework created by Facebook. It's fast, has great documentation, and provides snapshot testing capabilities.

Why Jest?

  • ‚úÖ Zero Config: Works with minimal setup
  • ‚úÖ Fast: Parallel test execution
  • ‚úÖ Great Matchers: Many built-in assertions
  • ‚úÖ Snapshot Testing: Detect unintended UI changes
  • ‚úÖ Mock Functions: Easy to test dependencies
  • ‚úÖ Coverage Reports: See which code is tested

Installation

# Install Jest via npm npm install --save-dev jest # Update package.json { "scripts": { "test": "jest", "test:watch": "jest --watch", "test:coverage": "jest --coverage" } } # Create your first test file: calculator.test.js # Jest auto-discovers files matching *.test.js or *.spec.js

4.2: Writing Your First Jest Test

// calculator.js - Function to test function add(a, b) { return a + b; } function multiply(a, b) { return a * b; } module.exports = { add, multiply };
// calculator.test.js - Test suite const { add, multiply } = require('./calculator'); describe('Calculator Functions', () => { test('should add two numbers correctly', () => { expect(add(2, 3)).toBe(5); }); test('should handle negative numbers', () => { expect(add(-5, 3)).toBe(-2); }); test('should multiply two numbers', () => { expect(multiply(4, 5)).toBe(20); }); }); // Run: npm test // Output: All tests pass ‚úì
Test Structure:
  • ‚Ä¢ describe(): Groups related tests
  • ‚Ä¢ test(): Individual test case
  • ‚Ä¢ expect(): The assertion
  • ‚Ä¢ .toBe(): Matcher (many more available)

4.3: Jest Matchers (Assertions)

// Equality expect(value).toBe(expected); // Exact match (===) expect(value).toEqual(expected); // Deep equality expect(value).not.toBe(expected); // Negation // Truthiness expect(value).toBeTruthy(); // Truthy value expect(value).toBeFalsy(); // Falsy value expect(value).toBeNull(); // null expect(value).toBeUndefined(); // undefined expect(value).toBeDefined(); // Defined // Numbers expect(value).toBeGreaterThan(5); expect(value).toBeLessThan(10); expect(value).toBeGreaterThanOrEqual(5); expect(value).toBeCloseTo(3.14, 1); // Approximate equality // Arrays & Objects expect(arr).toContain(item); expect(arr).toHaveLength(3); expect(obj).toHaveProperty('name'); expect(obj.name).toBe('John'); // Strings expect(str).toMatch(/regex/); expect(str).toHaveLength(5); // Exceptions expect(() => throwError()).toThrow(); expect(() => throwError()).toThrow(Error); expect(() => throwError()).toThrow('error message');

4.4: Setup and Teardown (Hooks)

describe('User Tests', () => { let user; // Runs before ALL tests in this suite beforeAll(() => { console.log('Starting user tests...'); }); // Runs before EACH test beforeEach(() => { user = { id: 1, name: 'John', email: 'john@example.com' }; }); // Runs after EACH test afterEach(() => { user = null; }); // Runs after ALL tests in this suite afterAll(() => { console.log('Finished all user tests'); }); test('should have user ID', () => { expect(user.id).toBe(1); }); test('should have user email', () => { expect(user.email).toContain('@'); }); });
When to Use Each Hook:
  • ‚Ä¢ beforeAll: Connect to database (do once)
  • ‚Ä¢ beforeEach: Create fresh test data (do before each test)
  • ‚Ä¢ afterEach: Clean up (delete test data, close connections)
  • ‚Ä¢ afterAll: Disconnect from database (do once)

4.5: Real Project - Login Validation

// login.js function validateLogin(email, password) { const errors = []; if (!email || !password) { errors.push('Email and password are required'); } if (email && !email.includes('@')) { errors.push('Invalid email format'); } if (password && password.length < 8) { errors.push('Password must be at least 8 characters'); } return { valid: errors.length === 0, errors: errors }; } module.exports = { validateLogin };
// login.test.js const { validateLogin } = require('./login'); describe('Login Validation', () => { test('valid credentials should pass', () => { const result = validateLogin('user@example.com', 'Password123'); expect(result.valid).toBe(true); expect(result.errors).toHaveLength(0); }); test('missing email should fail', () => { const result = validateLogin('', 'Password123'); expect(result.valid).toBe(false); expect(result.errors[0]).toContain('required'); }); test('invalid email format should fail', () => { const result = validateLogin('notanemail', 'Password123'); expect(result.valid).toBe(false); expect(result.errors).toContain('Invalid email format'); }); test('short password should fail', () => { const result = validateLogin('user@example.com', 'pass'); expect(result.valid).toBe(false); expect(result.errors[0]).toContain('at least 8 characters'); }); test('multiple errors should all be reported', () => { const result = validateLogin('invalid', 'short'); expect(result.errors).toHaveLength(2); }); });
