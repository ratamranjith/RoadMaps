5.1: Pytest Fundamentals

Pytest is a powerful Python testing framework. It's simple, scalable, and supports plugins. Perfect for both unit tests and integration tests.

Installation & Setup

# Install pytest pip install pytest # Create test file: test_calculator.py # Pytest auto-discovers files named test_*.py or *_test.py # Run tests pytest pytest -v # Verbose output pytest test_calculator.py # Specific file pytest -k "test_add" # Run specific tests pytest --tb=short # Short traceback pytest --lf # Run last failed pytest -x # Stop on first failure pytest --maxfail=3 # Stop after 3 failures

5.2: Writing Your First Pytest Test

# calculator.py def add(a, b): return a + b def multiply(a, b): return a * b def divide(a, b): if b == 0: raise ValueError("Cannot divide by zero") return a / b
# test_calculator.py from calculator import add, multiply, divide import pytest def test_add(): assert add(2, 3) == 5 def test_add_negative(): assert add(-5, 3) == -2 def test_multiply(): assert multiply(4, 5) == 20 def test_divide(): assert divide(10, 2) == 5 def test_divide_by_zero(): with pytest.raises(ValueError): divide(10, 0) # Run: pytest test_calculator.py # Output: 5 passed in 0.05s ‚úì
Test Naming Convention:
  • ‚Ä¢ Files: test_*.py or *_test.py
  • ‚Ä¢ Functions: test_*()
  • ‚Ä¢ Classes: Test*
  • ‚Ä¢ Assertions: Use assert statements

5.3: Fixtures (Setup & Teardown)

# test_user.py import pytest @pytest.fixture def user_data(): """Setup: Create test data""" data = { 'id': 1, 'name': 'John Doe', 'email': 'john@example.com' } print("\\nSetting up user data...") yield data print("\\nCleaning up user data...") def test_user_id(user_data): assert user_data['id'] == 1 def test_user_email(user_data): assert 'example.com' in user_data['email'] def test_user_name(user_data): assert user_data['name'] == 'John Doe' # Run: pytest test_user.py -v -s # Output: All 3 tests pass with setup/teardown messages
Fixture Types:
  • ‚Ä¢ function: Scope: each test (default)
  • ‚Ä¢ class: Scope: each class
  • ‚Ä¢ module: Scope: entire module
  • ‚Ä¢ session: Scope: entire test session

5.4: Parametrized Tests

Run the same test with multiple input values without duplicating code.

# test_parametrized.py import pytest @pytest.mark.parametrize("input,expected", [ (2, 4), (3, 9), (4, 16), (-2, 4), (0, 0), ]) def test_square(input, expected): def square(x): return x * x assert square(input) == expected # Multiple parameters @pytest.mark.parametrize("user,password,valid", [ ("john@example.com", "ValidPass123", True), ("john", "ValidPass123", False), ("john@example.com", "short", False), ("", "ValidPass123", False), ]) def test_login_validation(user, password, valid): def validate_login(email, pwd): return '@' in email and len(pwd) >= 8 assert validate_login(user, password) == valid # Run: pytest test_parametrized.py -v # Output: All parameter combinations tested

5.5: Real Project - API Testing

# test_api.py import pytest import requests @pytest.fixture def base_url(): return "https://jsonplaceholder.typicode.com" def test_get_user(base_url): """Test GET request""" response = requests.get(f"{base_url}/users/1") assert response.status_code == 200 data = response.json() assert data['id'] == 1 assert 'name' in data assert 'email' in data def test_create_post(base_url): """Test POST request""" payload = { "title": "Test Post", "body": "This is a test", "userId": 1 } response = requests.post(f"{base_url}/posts", json=payload) assert response.status_code == 201 data = response.json() assert data['title'] == "Test Post" @pytest.mark.parametrize("user_id", [1, 2, 3]) def test_get_multiple_users(base_url, user_id): """Test multiple users""" response = requests.get(f"{base_url}/users/{user_id}") assert response.status_code == 200 assert response.json()['id'] == user_id # Run: pytest test_api.py -v # Output: All API tests pass
